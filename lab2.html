Description

Task 1 requires the programmer to create a circuit that uses two buttons to control the way that the LED bar graph functions. Task 2 requires the programmer to have a button that changes the color of the neopixel and of the RGB light to the colors of the rainbow at the same time

Hardware components:

  • For Task 1:
    • GPIO Extension Board x1
    • ESP32-WROVER x1
    • Breadboard x1
    • LED bar graph x1
    • Jumper wire x8
    • push button x2
    • Resistor x12 at 220 ohm
  • For Task 2:
    • GPIO Extension Board x1
    • ESP32-WROVER x1
    • Breadboard x1
    • Jumper wire x9
    • Resistor x3 at 220 ohms
    • rgb LED x1
    • neopixel x1
    • Push button x1

Task 1:

            
from machine import Pin,PWM
from pwm import myPWM
import time

mypwm = myPWM(13,2,0,4,5,18,19,21)
chns=[0,1,2,3,4,5,6,7];
dutys=[0,0,0,0,0,0,0,0,1023,512,256,128,64,32,16,8,0,0,0,0,0,0,0,0];
delayTimes=50
button1 = Pin(15, Pin.IN, Pin.PULL_UP)
button2 = Pin(14, Pin.IN, Pin.PULL_UP)
def showled():
        for i in range(0,16):
            for j in range(0,8):
                mypwm.ledcWrite(chns[j],dutys[i+j])
            time.sleep_ms(delayTimes)
def showled2():           
        for i in range(0,16):
            for j in range(0,8):
                mypwm.ledcWrite(chns[7-j],dutys[i+j])
            time.sleep_ms(delayTimes)

try:
    while True:
        if not button1.value():
            showled2()
            
        if not button2.value():
            showled()
except:
    mypwm.deinit()

Task 2:

            
from machine import Pin,PWM
import time

import neopixel
pin = Pin(5, Pin.OUT)
np = neopixel.NeoPixel(pin, 8)

pins=[15,2,0] #red(15) #green(2) #blue(0)
button = Pin(14, Pin.IN, Pin.PULL_UP)

pwm0=PWM(Pin(pins[0]),10000)
pwm1=PWM(Pin(pins[1]),10000)
pwm2=PWM(Pin(pins[2]),10000)

def setColor(r,g,b):
    pwm0.duty(1023-r)
    pwm1.duty(1023-g)
    pwm2.duty(1023-b)
  
color = [[255, 0, 0],[255, 50, 0],[255,255,0],[0,255,0],[0,0,255],[15, 0, 220],[25,0,200]]
color2 = [[20, 0, 0], [20, 4, 0], [20, 20, 0], [0, 20, 0], [0, 0, 20], [6, 0, 10], [10, 0, 20]] 

index = 0
index2 = 0

c=len(color)
c2=len(color2)

# Start Here
try:
    while True:
        if button.value() == 0:
            
            r, g, b = color[index]
            setColor(b, g, r)
            time.sleep_ms(200)
            index = (index + 1) % c

            current_color = color2[index2]
            for j in range(0, 8):
                np[j] = current_color
            np.write()
            time.sleep_ms(50)
            index2 = (index2 + 1) % c2
            time.sleep_ms(500)
            
except:
    pwm0.deinit()
    pwm1.deinit()
    pwm2.deinit()
