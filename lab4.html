from machine import TouchPad, Pin, PWM
import time

PRESS_VAL=150
PRESS_VAL2=170

buzzer = PWM(Pin(2))
led=Pin(23,Pin.OUT)
tp = TouchPad(Pin(4,Pin.IN,Pin.PULL_UP))
led2=Pin(21,Pin.OUT)
tp2 = TouchPad(Pin(15,Pin.IN,Pin.PULL_UP))
led3=Pin(27,Pin.OUT)
tp3 = TouchPad(Pin(13,Pin.IN,Pin.PULL_UP))
led4=Pin(22,Pin.OUT)
tp4 = TouchPad(Pin(12,Pin.IN,Pin.PULL_UP))
led5=Pin(19,Pin.OUT)
tp5 = TouchPad(Pin(14,Pin.IN,Pin.PULL_UP))
led6=Pin(26,Pin.OUT)
tp6 = TouchPad(Pin(33,Pin.IN,Pin.PULL_UP))
led7=Pin(25,Pin.OUT)
tp7 = TouchPad(Pin(32,Pin.IN,Pin.PULL_UP))

C4  = 262  # C
D4  = 294  # D
E4  = 330  # E
F4  = 349  # F
G4  = 392  # G
A4  = 440  # A
B4  = 494  # B


def playTone(N):
    buzzer.freq(N)
    buzzer.duty(512)
    time.sleep(0.2)
    buzzer.duty(0)
    
while True:
    if tp.read() < PRESS_VAL:
        led.value(1)
        playTone(C4)
        led.value(0)
    if tp2.read() < PRESS_VAL2:
        led2.value(1)
        playTone(D4)
        led2.value(0)
    if tp3.read() < PRESS_VAL2:
        led3.value(1)
        playTone(E4)
        led3.value(0)
    if tp4.read() < PRESS_VAL2:
        led4.value(1)
        playTone(F4)
        led4.value(0)
    if tp5.read() < PRESS_VAL:
        led5.value(1)
        playTone(G4)
        led5.value(0)
    if tp6.read() < PRESS_VAL2:
        led6.value(1)
        playTone(A4)
        led6.value(0)
    if tp7.read() < PRESS_VAL:
        led7.value(1)
        playTone(B4)
        led7.value(0)

from machine import Pin,PWM,ADC
import time
import neopixel
import random

button=Pin(13,Pin.IN,Pin.PULL_UP)
activeBuzzer=Pin(12,Pin.OUT)
activeBuzzer.value(0)

pin = Pin(2, Pin.OUT)
np = neopixel.NeoPixel(pin, 8)

def crazyBuzzer():
    activeBuzzer.value(1)
    time.sleep_ms(100)
    activeBuzzer.value(0)
    time.sleep_ms(100)
   
def random_color():
    red = random.randint(0, 255)
    green = random.randint(0, 255)
    blue = random.randint(0, 255)
    return (red, green, blue)

pwm =PWM(Pin(25,Pin.OUT),1000)
adc=ADC(Pin(36))
adc.atten(ADC.ATTN_11DB)
adc.width(ADC.WIDTH_10BIT)

try:
    while True:
        adcValue=adc.read()
        pwm.duty(adcValue)
        dimmer = 1023 - adcValue
        pwm.duty(dimmer)
        print(adc.read())
        time.sleep_ms(100)
        
        if adcValue < 550:
            crazyBuzzer()
            color = random_color()  
            for i in range(8):
                np[i] = color  
            np.write()
            time.sleep_ms(200)
            
        if not button.value():
            for i in range(len(np)):
                np[i] = (0, 0, 0)
            np.write()
            break           
except:
    pwm.deinit()
